js 获取当前月份的开始日期和结束日期

//  获取月份开始日期和结束日期

function getMonthStartEndDate(year, month) {
    let d = new Date(year, month, 0);
    let days = d.getDate();
    return {
        startDate: year + "-" + month + "-01",
        endDate: year + "-" + month + "-" + days,
    };
}

console.log(getMonthStartEndDate(2022, 1));

console.log(getMonthStartEndDate('2022', '08'));
